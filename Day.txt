# Example: Fetch and display GCP instance details using subprocess
import subprocess

def get_instance_details(instance_name, zone):
    """Fetches details for a GCP Compute Engine instance."""
    try:
        result = subprocess.run(
            [
                "gcloud", "compute", "instances", "describe",
                instance_name, "--zone", zone, "--format=json"
            ],
            capture_output=True, text=True, check=True
        )
        print("Instance details:\n", result.stdout)
    except subprocess.CalledProcessError as e:
        print("Error fetching instance details:", e.stderr)

if __name__ == "__main__":
    # Change these to your instance name and zone
    get_instance_details("instance-20250902-195404", "us-central1-c")
